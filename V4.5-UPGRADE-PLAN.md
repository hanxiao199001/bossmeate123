### A-2. 创建 SkillRegistry

```typescript
import type { ISkill } from "./base-skill.js";

class SkillRegistryClass {
  private skills = new Map<string, ISkill>();

  /** 注册一个技能 */
  register(skill: ISkill): void {
    if (this.skills.has(skill.name)) {
      throw new Error(`Skill "${skill.name}" already registered`);
    }
    this.skills.set(skill.name, skill);
    console.log(`✅ Skill registered: ${skill.name} (${skill.displayName})`);
  }

  /** 根据 skillType 获取对应 Skill 实例 */
  get(name: string): ISkill | undefined {
    return this.skills.get(name);
  }

  /** 获取所有已注册技能列表（用于前端展示） */
  list(): Array<{
    name: string;
    displayName: string;
    description: string;
  }> {
    return Array.from(this.skills.values()).map((s) => ({
      name: s.name,
      displayName: s.displayName,
      description: s.description,
    }));
  }

  /** 检查某技能是否已注册 */
  has(name: string): boolean {
    return this.skills.has(name);
  }
}

/** 全局单例 */
export const SkillRegistry = new SkillRegistryClass();
```

### A-3. 改造 ArticleSkill 实现 ISkill 接口

**修改文件**: `packages/server/src/services/skills/article-skill.ts`

2. 让 `ArticleSkill` 类实现 `ISkill` 接口，新增以下属性和方法（不删除任何已有方法）：

```typescript
export class ArticleSkill implements ISkill {
  // ===== ISkill 接口实现 =====
  readonly name = "article";
  readonly displayName = "智能图文";
  readonly description = "基于 RAG 知识库的学术/行业图文生成，含自动质量检查";
  readonly preferredTier = "expensive" as const;

  private provider: AIProvider;

  constructor(provider: AIProvider) {
    this.provider = provider;
  }

  /**
   * ISkill.handle 方法 —— 统一入口
   * 内部复用已有的 understandRequirement / fullGenerate / qualityCheck
   */
  async handle(
    userInput: string,
    history: ChatMessage[],
    context: SkillContext
  ): Promise<SkillResult> {
    // 第一步：理解需求（复用已有方法）
    const { parsed, response } = await this.understandRequirement(userInput, history);

    // 如果需要澄清，直接返回追问
    if (parsed.needsClarification) {
      return {
        reply: response,
        tokenUsage: undefined, // 追问阶段不计 token
      };
    }

    // 第二步：RAG 检索（如果 context 没有预提供 ragContext）
    let ragText = context.ragContext;
    if (!ragText) {
      try {
        const ragResult = await retrieveForArticle({
          tenantId: context.tenantId,
          topic: parsed.topic,
          audience: parsed.audience,
          tone: parsed.tone,
          keywords: parsed.keyPoints,
        });
        ragText = ragResult.text;
      } catch (err) {
        console.warn("RAG retrieval failed, proceeding without context:", err);
      }
    }

    // 第三步：生成 + 质量检查（复用已有方法）
    const { article, quality } = await this.fullGenerate(parsed, ragText);

    return {
      reply: response,
      artifact: {
        type: "article",
        title: article.title,
        body: article.body,
        summary: article.summary,
        tags: article.tags,
        metadata: {
          wordCount: article.wordCount,
          qualityScore: quality.score,
          qualityPassed: quality.passed,
          issues: quality.issues,
          suggestions: quality.suggestions,
        },
      },
    };
  }

  // ===== 以下保留原有方法不变 =====
  // understandRequirement(...)
  // generateArticle(...)
  // qualityCheck(...)
  // fullGenerate(...)
}
```

## 模块 B：Task 系统（BullMQ 异步任务）

### 目标

将耗时操作（文章生成、批量处理、定时爬虫）从同步 HTTP 请求改为后台异步任务，支持进度推送和失败重试。

### B-1. 安装 BullMQ

```bash
cd packages/server && pnpm add bullmq
```

注意：BullMQ 依赖 ioredis，已安装，不需要额外操作。

### B-2. 创建 Redis 连接和队列管理

**新建文件**: `packages/server/src/services/task/queue.ts`

```typescript
import { Queue, Worker, QueueEvents, Job } from "bullmq";
import IORedis from "ioredis";
import { env } from "../../config/env.js";

// ===== Redis 连接（复用同一实例） =====
let connection: IORedis | null = null;

export function getRedisConnection(): IORedis {
  if (!connection) {
    connection = new IORedis(env.REDIS_URL, {
      maxRetriesPerRequest: null, // BullMQ 要求
    });
    connection.on("error", (err) => {
      console.error("❌ Redis connection error:", err.message);
    });
    connection.on("connect", () => {
      console.log("✅ Redis connected for task queue");
    });
  }
  return connection;
}

// ===== 队列定义 =====

/** 内容生成队列 —— 处理文章/视频脚本等耗时生成 */
export const contentQueue = new Queue("content-generation", {
  connection: getRedisConnection(),
  defaultJobOptions: {
    attempts: 2,           // 最多重试 2 次
    backoff: {
      type: "exponential",
      delay: 5000,         // 首次重试等 5 秒
    },
    removeOnComplete: 100, // 保留最近 100 条完成记录
    removeOnFail: 50,      // 保留最近 50 条失败记录
  },
});

/** 爬虫队列 —— 替代原来 scheduler.ts 里的 setTimeout */
export const crawlerQueue = new Queue("crawler", {
  connection: getRedisConnection(),
  defaultJobOptions: {
    attempts: 3,
    backoff: { type: "exponential", delay: 10000 },
    removeOnComplete: 50,
    removeOnFail: 50,
  },
});

// ===== 队列事件（可选：用于 WebSocket 推送） =====
export const contentQueueEvents = new QueueEvents("content-generation", {
  connection: getRedisConnection(),
});

/** 关闭所有连接（graceful shutdown） */
export async function closeQueues(): Promise<void> {
  await contentQueue.close();
  await crawlerQueue.close();
  await contentQueueEvents.close();
  if (connection) {
    await connection.quit();
    connection = null;
  }
}
```

### B-4. 创建内容生成 Worker

**新建文件**: `packages/server/src/services/task/content-worker.ts`

```typescript
import { Worker, Job } from "bullmq";
import { getRedisConnection } from "./queue.js";
import { SkillRegistry } from "../skills/index.js";
import { getProvider } from "../ai/provider-factory.js";
import { db } from "../../models/db.js"; // 你的 pg client

/** 任务输入结构 */
interface ContentJobData {
  taskId: string;
  tenantId: string;
  userId: string;
  conversationId: string;
  skillType: string;
  userInput: string;
  history: Array<{ role: string; content: string }>;
}

/** 启动内容生成 Worker */
export function startContentWorker(): Worker {
  const worker = new Worker<ContentJobData>(
    "content-generation",
    async (job: Job<ContentJobData>) => {
      const { taskId, tenantId, userId, conversationId, skillType, userInput, history } = job.data;

      // 更新任务状态为 running
      await db.query(
        `UPDATE tasks SET status = 'running', started_at = NOW(), updated_at = NOW() WHERE id = $1`,
        [taskId]
      );

      const skill = SkillRegistry.get(skillType);
      if (!skill) {
        throw new Error(`Skill "${skillType}" not found in registry`);
      }

      const provider = getProvider(skill.preferredTier) || getProvider("cheap");
      if (!provider) {
        throw new Error("No AI provider available");
      }

      // 更新进度: 开始处理
      await job.updateProgress(10);
      await db.query(`UPDATE tasks SET progress = 10, updated_at = NOW() WHERE id = $1`, [taskId]);

      // 记录开始日志
      const stepStart = Date.now();

      // 调用 Skill 处理
      const result = await skill.handle(userInput, history as any, {
        tenantId,
        userId,
        conversationId,
        provider,
      });

      // 更新进度: 生成完成
      await job.updateProgress(90);
      await db.query(`UPDATE tasks SET progress = 90, updated_at = NOW() WHERE id = $1`, [taskId]);

      // 记录执行日志
      await db.query(
        `INSERT INTO task_logs (task_id, step, status, model, duration_ms, detail)
         VALUES ($1, 'skill_handle', 'completed', $2, $3, $4)`,
        [
          taskId,
          provider.name,
          Date.now() - stepStart,
          JSON.stringify({
            hasArtifact: !!result.artifact,
            tokenUsage: result.tokenUsage,
          }),
        ]
      );

      // 如果有制品，保存到 contents 表
      if (result.artifact) {
        await db.query(
          `INSERT INTO contents (tenant_id, user_id, conversation_id, type, title, body, status, metadata)
           VALUES ($1, $2, $3, $4, $5, $6, 'draft', $7)`,
          [
            tenantId, userId, conversationId,
            result.artifact.type,
            result.artifact.title,
            result.artifact.body,
            JSON.stringify(result.artifact.metadata || {}),
          ]
        );
      }

      // 更新任务状态为完成
      await db.query(
        `UPDATE tasks SET status = 'completed', progress = 100, output = $2,
         completed_at = NOW(), updated_at = NOW() WHERE id = $1`,
        [taskId, JSON.stringify({ reply: result.reply, artifact: result.artifact })]
      );

      return { reply: result.reply, hasArtifact: !!result.artifact };
    },
    {
      connection: getRedisConnection(),
      concurrency: 3, // 最多同时处理 3 个生成任务
    }
  );

  worker.on("completed", (job) => {
    console.log(`✅ Task completed: ${job.id}`);
  });

  worker.on("failed", async (job, err) => {
    console.error(`❌ Task failed: ${job?.id}`, err.message);
    if (job) {
      await db.query(
        `UPDATE tasks SET status = 'failed', error = $2, updated_at = NOW() WHERE id = $1`,
        [job.data.taskId, err.message]
      );
    }
  });

  console.log("🔄 Content generation worker started (concurrency: 3)");
  return worker;
}
```

### B-5. 创建任务路由

**新建文件**: `packages/server/src/routes/tasks.ts`

```typescript
import type { FastifyInstance } from "fastify";
import { z } from "zod";
import { contentQueue } from "../services/task/queue.js";
import { db } from "../models/db.js";

const createTaskSchema = z.object({
  conversationId: z.string().uuid(),
  skillType: z.string(),
  userInput: z.string(),
  history: z.array(z.object({
    role: z.string(),
    content: z.string(),
  })).optional().default([]),
});

export default async function taskRoutes(fastify: FastifyInstance) {
  // POST /api/v1/tasks — 创建异步任务
  fastify.post("/", async (request, reply) => {
    const body = createTaskSchema.parse(request.body);
    const tenantId = request.tenantId;
    const userId = request.user.id;

    // 1. 创建 task 记录
    const result = await db.query(
      `INSERT INTO tasks (tenant_id, user_id, conversation_id, type, status, input)
       VALUES ($1, $2, $3, $4, 'pending', $5) RETURNING id`,
      [
        tenantId, userId, body.conversationId,
        `${body.skillType}_generation`,
        JSON.stringify({ userInput: body.userInput }),
      ]
    );
    const taskId = result.rows[0].id;

    // 2. 加入队列
    await contentQueue.add(`gen-${taskId}`, {
      taskId,
      tenantId,
      userId,
      conversationId: body.conversationId,
      skillType: body.skillType,
      userInput: body.userInput,
      history: body.history,
    });

    return reply.status(202).send({
      taskId,
      status: "pending",
      message: "任务已提交，正在后台处理",
    });
  });

  // GET /api/v1/tasks/:id — 查询任务状态
  fastify.get("/:id", async (request, reply) => {
    const { id } = request.params as { id: string };
    const tenantId = request.tenantId;

    const result = await db.query(
      `SELECT id, type, status, progress, output, error, created_at, started_at, completed_at
       FROM tasks WHERE id = $1 AND tenant_id = $2`,
      [id, tenantId]
    );

    if (result.rows.length === 0) {
      return reply.status(404).send({ error: "任务不存在" });
    }

    return result.rows[0];
  });

  // GET /api/v1/tasks — 查询当前用户的任务列表
  fastify.get("/", async (request, reply) => {
    const tenantId = request.tenantId;
    const userId = request.user.id;
    const { status, limit = 20 } = request.query as { status?: string; limit?: number };

    let sql = `SELECT id, type, status, progress, error, created_at, completed_at
               FROM tasks WHERE tenant_id = $1 AND user_id = $2`;
    const params: any[] = [tenantId, userId];

    if (status) {
      sql += ` AND status = $${params.length + 1}`;
      params.push(status);
    }

    sql += ` ORDER BY created_at DESC LIMIT $${params.length + 1}`;
    params.push(limit);

    const result = await db.query(sql, params);
    return { tasks: result.rows };
  });

  // GET /api/v1/tasks/:id/logs — 查询任务执行日志
  fastify.get("/:id/logs", async (request, reply) => {
    const { id } = request.params as { id: string };
    const tenantId = request.tenantId;

    // 先验证任务归属
    const task = await db.query(
      `SELECT id FROM tasks WHERE id = $1 AND tenant_id = $2`,
      [id, tenantId]
    );
    if (task.rows.length === 0) {
      return reply.status(404).send({ error: "任务不存在" });
    }

    const logs = await db.query(
      `SELECT step, status, input_tokens, output_tokens, model, duration_ms, detail, created_at
       FROM task_logs WHERE task_id = $1 ORDER BY created_at ASC`,
      [id]
    );

    return { logs: logs.rows };
  });

  // POST /api/v1/tasks/:id/cancel — 取消任务
  fastify.post("/:id/cancel", async (request, reply) => {
    const { id } = request.params as { id: string };
    const tenantId = request.tenantId;

    const result = await db.query(
      `UPDATE tasks SET status = 'cancelled', updated_at = NOW()
       WHERE id = $1 AND tenant_id = $2 AND status IN ('pending', 'running')
       RETURNING id`,
      [id, tenantId]
    );

    if (result.rows.length === 0) {
      return reply.status(400).send({ error: "任务不存在或无法取消" });
    }

    return { message: "任务已取消" };
  });
}
```

### B-6. WebSocket 任务进度推送

**新建文件**: `packages/server/src/services/task/progress-ws.ts`

```typescript
import type { FastifyInstance } from "fastify";
import { contentQueueEvents } from "./queue.js";

/**
 * 注册 WebSocket 路径 /ws/tasks/:taskId
 * 客户端连接后实时接收任务进度更新
 */
export function registerTaskWebSocket(fastify: FastifyInstance): void {
  fastify.get("/ws/tasks/:taskId", { websocket: true }, (socket, request) => {
    const { taskId } = request.params as { taskId: string };

    // 监听进度事件
    const onProgress = ({ jobId, data }: { jobId: string; data: number | object }) => {
      // BullMQ job name 格式: gen-{taskId}
      if (jobId === `gen-${taskId}` || String(data).includes(taskId)) {
        socket.send(JSON.stringify({
          type: "progress",
          taskId,
          progress: typeof data === "number" ? data : 0,
        }));
      }
    };

    const onCompleted = ({ jobId, returnvalue }: { jobId: string; returnvalue: string }) => {
      if (jobId === `gen-${taskId}`) {
        socket.send(JSON.stringify({
          type: "completed",
          taskId,
          result: JSON.parse(returnvalue || "{}"),
        }));
        socket.close();
      }
    };

    const onFailed = ({ jobId, failedReason }: { jobId: string; failedReason: string }) => {
      if (jobId === `gen-${taskId}`) {
        socket.send(JSON.stringify({
          type: "failed",
          taskId,
          error: failedReason,
        }));
        socket.close();
      }
    };

    contentQueueEvents.on("progress", onProgress);
    contentQueueEvents.on("completed", onCompleted);
    contentQueueEvents.on("failed", onFailed);

    // 客户端断开时清理监听
    socket.on("close", () => {
      contentQueueEvents.off("progress", onProgress);
      contentQueueEvents.off("completed", onCompleted);
      contentQueueEvents.off("failed", onFailed);
    });
  });
}
```

### C-1. 定义 Tool 接口

**新建文件**: `packages/server/src/services/tools/base-tool.ts`

```typescript
import { z } from "zod";

/** Tool 执行上下文 */
export interface ToolContext {
  tenantId: string;
  userId: string;
}

/** Tool 执行结果 */
export interface ToolResult {
  success: boolean;
  data?: unknown;
  error?: string;
  /** 本次调用的耗时（毫秒） */
  durationMs?: number;
}

/** 所有 Tool 的标准接口 */
export interface ITool {
  /** 工具唯一名称（英文，snake_case） */
  readonly name: string;
  /** 工具描述（给 AI 看的，用于决定是否调用此工具） */
  readonly description: string;
  /** 参数 schema（Zod，用于校验 + 生成 JSON Schema 给 AI） */
  readonly parameters: z.ZodObject<any>;

  /** 执行工具 */
  execute(params: Record<string, unknown>, context: ToolContext): Promise<ToolResult>;
}

/**
 * 把 Zod schema 转为 JSON Schema（简化版）
 * 用于将来传给 AI 的 function calling
 */
export function zodToJsonSchema(schema: z.ZodObject<any>): Record<string, unknown> {
  // 简化实现：遍历 shape 提取类型描述
  const shape = schema.shape;
  const properties: Record<string, unknown> = {};
  const required: string[] = [];

  for (const [key, value] of Object.entries(shape)) {
    const field = value as z.ZodTypeAny;
    // 基本类型映射
    if (field instanceof z.ZodString) {
      properties[key] = { type: "string", description: field.description };
    } else if (field instanceof z.ZodNumber) {
      properties[key] = { type: "number", description: field.description };
    } else if (field instanceof z.ZodBoolean) {
      properties[key] = { type: "boolean", description: field.description };
    } else if (field instanceof z.ZodArray) {
      properties[key] = { type: "array", description: field.description };
    } else {
      properties[key] = { type: "string" };
    }

    if (!field.isOptional()) {
      required.push(key);
    }
  }

  return { type: "object", properties, required };
}
```
